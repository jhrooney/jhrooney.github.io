crime_cols_to_use = ["Date Rptd", "Crm Cd Desc", "LAT", "LON"]
crime = pd.read_csv("la_crime_data.csv", usecols=crime_cols_to_use)

def convert_date(df, col):
    return pd.to_datetime(df[col], format="%m/%d/%Y %I:%M:%S %p")

crime_col_renaming = {
    "Crm Cd Desc": "crime_description",
    "LAT": "latitude",
    "LON": "longitude",
}

def clean_data(df):
    return (
        df
        .assign(date_reported=convert_date(df, "Date Rptd"))
        .rename(columns=crime_col_renaming)
        .astype({"crime_description": "category"})
        .drop(columns="Date Rptd")
    )

crime = clean_data(crime)
crime = crime.loc[(crime.latitude != 0) | (crime.longitude != 0)]
def rand_shift(num_items, magnitude):
    result = ((np.random.random(num_items) / 2) + 0.5) * 2 * magnitude / 6378
    return np.where(np.random.random(num_items) > 0.5, result, -result)

np.random.seed(42)
crime.loc[:, "latitude"] = crime["latitude"] + rand_shift(len(crime), 0.5)
crime.loc[:, "longitude"] = crime["longitude"] + rand_shift(len(crime), 0.5)

def filter_year(df, year):
    return df.loc[df.date_reported.dt.year == year]

def agg_crimes_by_year(df, year):
    filtered_df = filter_year(df, year)
    data_to_concat = [
        census_tracts,
        filtered_df.groupby("tract").size().rename("total_crime")
    ] + [
        filtered_df
        .loc[filtered_df.crime_description.isin(v)]
        .groupby("tract")
        .size()
        .rename(k)
        for k, v in crime_classes.items()
    ]
    return (
        pd.concat(data_to_concat, axis="columns")
        .fillna(0)
        .astype({k: "int64" for k in crime_classes})
    )

    "drug_alcohol": arrests_drug_alcohol,
    "weapon": arrests_weapon,
    "sex": arrests_sex,
    "fraud": arrests_fraud,
}
def filter_year_arrests(df, year):
    return df.loc[df.arrest_date.dt.year == year]

def agg_arrests_by_year(df, year):
    filtered_df = filter_year_arrests(df, year)
    data_to_concat = [
        census_tracts,
        filtered_df.groupby("tract").size().rename("total_arrests")
    ] + [
        filtered_df
        .loc[filtered_df.charge_description.isin(v)]
        .groupby("tract")
        .size()
        .rename(k)
        for k, v in arrest_classes.items()
    ]
    return (
        pd.concat(data_to_concat, axis="columns")
        .fillna(0)
        .astype({k: "int64" for k in {"total_arrests"} | arrest_classes.keys()})
    )
arrests_by_year = {
    str(year): agg_arrests_by_year(arrests_by_tract, year=year)
    for year in available_years
}
